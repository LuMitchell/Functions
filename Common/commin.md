```
$query = app_db::query("SELECT name,areaid,company,province,amount,adviceprice,price,limitprice,sms,rechargetime,typeid from product where type='mobile' and pricetype='min'  ORDER BY --amount,areaid desc");
while($row = app_db::fetch_one($query))
{
    if(in_array($row['company'], ['移动', '联通', '电信']) && $row['province'] == '全国') continue;

    $minprice = round(floatval($rd->hget('mobileprice', $row['typeid'])), 3);
    if($minprice < 0) continue;

    $row['price'] = sprintf('%.3f', $minprice);
    unset($row['typeid']);
    $arr = explode('话费', $row['name']);

    preg_match('#(\d+)元#is', $arr[1], $matches);
    $amount = $matches[1];

    $lists[$row['province']][$row['company']][$amount] = $row['price'];
}

app_tpl::assign('lists', $lists, $path_tpls_main);
```
